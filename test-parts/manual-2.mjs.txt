async function stalkerScenarios() {
  async function encounter(preference, patch={}, randomValue=null) {
    const {page,errors}=await newPage();
    await click(page,/Новая смена/);
    await page.evaluate(()=>window.__zeroShiftTest.startChapter(3));
    if(randomValue!==null) await page.evaluate(v=>{Math.random=()=>v;},randomValue);
    await page.evaluate(p=>window.__zeroShiftTest.forceState({...p, threatRoom:'ringA', threatFrozen:1}),patch);
    const before=await state(page);
    await move(page,'восток');
    const choices=await choiceTexts(page);
    assert(choices.some(t=>/Имитатор|Заблокировать|оглушающий|Замереть|Рвануть/.test(t)),'Преследователь вызывает полноценное столкновение');
    if(preference==='lock') {
      await click(page,/Заблокировать переборку/); await continueAction(page);
      assert((await state(page)).battery===before.battery-1,'Блокировка тратит батарею');
    } else if(preference==='stun') {
      await click(page,/оглушающий импульс/); await continueAction(page);
      assert((await state(page)).stun===before.stun-1,'Оглушение тратит импульс');
    } else if(preference==='hide') {
      await click(page,/Замереть и спрятаться/); await continueAction(page);
      const after=await state(page);
      assert(randomValue>.38 ? after.health===before.health : after.health<before.health,'Укрытие учитывает успех и провал');
    } else if(preference==='flee') {
      await click(page,/Рвануть/);
      const after=await state(page);
      assert(after.roomId!=='ringA' && after.health<before.health,'Побег меняет комнату и наносит урон');
    }
    assert(errors.length===0,`Столкновение ${preference} без ошибок JavaScript`,errors.join('\n'));
    await page.close();
  }
  await encounter('lock');
  await encounter('stun',{battery:0,stun:2});
  await encounter('hide',{battery:0,stun:0},0.9);
  await encounter('hide',{battery:0,stun:0},0.1);
  await encounter('flee',{battery:0,stun:0});
}

    return all.filter((id,i)=>all.indexOf(id)!==i);
